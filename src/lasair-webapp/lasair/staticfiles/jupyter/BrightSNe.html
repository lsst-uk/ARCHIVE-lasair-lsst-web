def plotLC(lightcurve):
    mjd_now = time.time()/86400 + 40587.0
    color = {1:'green', 2:'red'}
    for fid in [1,2]:
        since = []
        magpsf = []
        sigmapsf = []
        for cand in lightcurve:
            if cand['fid'] == fid:
                since.append(cand['mjd'] - mjd_now)
                magpsf.append(cand['magpsf'])
                sigmapsf.append(cand['sigmapsf'])
        plt.errorbar(since, magpsf, yerr=sigmapsf, c=color[fid])
        plt.xlabel("days before %.2f" % mjd_now)
        plt.ylabel("psf mag")
        plt.title('https://lasair-iris.roe.ac.uk/objects/%s' % objectId)
        plt.gca().invert_yaxis()
    plt.show()
